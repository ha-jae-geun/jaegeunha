# 재귀함수
```C
#include <stdio.h>
#include<math.h>
int recursive(int n) {
   int i;
   if (n < 1) { return 2; }
   else
   {
      i = (2 * recursive(n - 1)) + 1;
      printf("%d\n", i);
      return i;
   }
}




void main() {
   int i;
   printf("숫자를 입력하시오");
   scanf_s("%d", &i);
   recursive(i);


}
//5 11 23 47 95
``` 
