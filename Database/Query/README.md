2.
```SQL
SELECT   SIDO,
         SUBSTR(MEASURE_DATE, 0, 8) mea_date,
         ROUND(AVG(SLFR_DXD), 6),
         ROUND(AVG(CRBN_MNXD), 6),
         ROUND(AVG(NVL(OZON,0)), 6),
         ROUND(AVG(NTRGN_DXD), 6),
         ROUND(AVG(PARTICLE_MATTER_10), 6),
         ROUND(AVG(PARTICLE_MATTER_2), 6)
FROM     MICRO_DUST_2017
WHERE    SIDO = '경기'
GROUP BY SIDO,
         SUBSTR(MEASURE_DATE, 0, 8)
```

3.
```SQL
SELECT   SIDO,
         SUBSTR(MEASURE_DATE, 0, 8) mea_date,
         ROUND(AVG(SLFR_DXD), 6),
         ROUND(AVG(CRBN_MNXD), 6),
         ROUND(AVG(OZON), 6),
         ROUND(AVG(NTRGN_DXD), 6),
         ROUND(AVG(PARTICLE_MATTER_10), 6),
         ROUND(AVG(PARTICLE_MATTER_2), 6)
FROM     MICRO_DUST_2017
WHERE    SIDO = '경기'

GROUP BY SIDO,
         SUBSTR(MEASURE_DATE, 0, 8)
```

4.
```SQL
SELECT SUBSTR(B.BASE_NAME,0,2),
       REPLACE(A.MEA_DATE, '-', ''),
       A.AVG_TMPRT,
       A.DAY_PRE,
       A.AVG_WND_SPD,
       A.WND_DRCTN,
       A.TOT_DYL_TM,
       A.TOT_SOLAR
       
FROM   WEATHER_2015 A,
       TB_BASE_CODE B
WHERE  B.KIND_CODE    = 1
AND    A.AREA (+)     = B.BASE_CODE
AND    B.BASE_NAME LIKE '서울%'
```

6.
```SQ;
SELECT SUBSTR(B.BASE_NAME,0,4) SIDO,
       REPLACE(A.MEA_DATE, '-', '') MEAS_DATE,
       ROUND(AVG(A.AVG_TMPRT), 3),
       ROUND(AVG(A.DAY_PRE), 3),
       ROUND(AVG(A.AVG_WND_SPD), 3),
       ROUND(AVG(A.WND_DRCTN), 3),
       ROUND(AVG(A.TOT_DYL_TM), 3),
       ROUND(AVG(A.TOT_SOLAR), 3)  
FROM   COPY_WEATHER_2015 A,
       TB_BASE_CODE B
WHERE  B.KIND_CODE    = 1
AND    A.AREA (+)     = B.BASE_CODE
GROUP BY SUBSTR(B.BASE_NAME,0,4), REPLACE(A.MEA_DATE, '-', '')
ORDER BY SIDO, MEAS_DATE
```
